sudo apt-get update
sudo apt install cmake build-essential libsndfile1 libsndfile1-dev libboost-all-dev htop net-tools debhelper devscripts -y

sudo apt --fix-broken install
sudo apt-get install lockfile-progs

git clone https://github.com/rbrito/usbmount.git
cd usbmount
sudo dpkg -i ../usbmount_*.deb

mkdir build
cd build
cmake -DCMAKE_BUILD_TYPE=Release ..
make -j4

sudo nano /etc/systemd/system/RavennaRecorder.service

[Unit]
Description=RavennaRecorder
[Service]
Environment="HOME=/home/pi"
ExecStartPre=/bin/sleep 3
ExecStart=/home/pi/RavennaRecorder/build/RavennaRecorder
Restart=always
User=root
Group=root
Environment=PATH=/usr/bin:/usr/local/bin
[Install]
WantedBy=multi-user.target

sudo systemctl daemon-reload
sudo systemctl enable RavennaRecorder
sudo systemctl start RavennaRecorder
sudo systemctl status RavennaRecorder
